    ניהול תיבות דואר ב־Exchange Online PowerShell

    🔍 צפייה בתיבות דואר

    # הצגת כל תיבות הדואר בארגון
Get-Mailbox

# הצגת תיבת דואר אחת
Get-Mailbox -Identity "user@domain.com"

# הצגת תיבות דואר מסוג Room/Shared
Get-Mailbox -RecipientTypeDetails RoomMailbox
Get-Mailbox -RecipientTypeDetails SharedMailbox

    ✏️ שינוי תכונות בתיבת דואר

    # שינוי שם התצוגה
Set-Mailbox -Identity "user@domain.com" -DisplayName "New Display Name"

# הפיכת תיבת דואר רגילה ל־Shared
Set-Mailbox -Identity "user@domain.com" -Type Shared

    📬 הרשאות על תיבות דואר

    # צפייה בהרשאות גישה לתיבת דואר
Get-MailboxPermission -Identity "user@domain.com"

# הוספת הרשאת Full Access
Add-MailboxPermission -Identity "user@domain.com" -User "otheruser@domain.com" -AccessRights FullAccess -InheritanceType All

# הסרת הרשאת Full Access
Remove-MailboxPermission -Identity "user@domain.com" -User "otheruser@domain.com" -AccessRights FullAccess

    📆 הרשאות ליומן (Calendar)

    # צפייה בהרשאות ליומן
Get-MailboxFolderPermission -Identity "user@domain.com:\Calendar"

# הוספת הרשאת צפייה ביומן
Add-MailboxFolderPermission -Identity "user@domain.com:\Calendar" -User "otheruser@domain.com" -AccessRights Reviewer

# הסרת הרשאה מהיומן
Remove-MailboxFolderPermission -Identity "user@domain.com:\Calendar" -User "otheruser@domain.com"

    🔄 ניהול העברה (Forwarding)

    # העברת מיילים למשתמש אחר
Set-Mailbox -Identity "user@domain.com" -ForwardingSMTPAddress "forwardto@domain.com"

# ביטול העברה
Set-Mailbox -Identity "user@domain.com" -ForwardingSMTPAddress $null

    📤 הגדרת Send As / Send on Behalf

    # הרשאת Send As
Add-RecipientPermission -Identity "shared@domain.com" -Trustee "user@domain.com" -AccessRights SendAs

# הרשאת Send on Behalf
Set-Mailbox -Identity "shared@domain.com" -GrantSendOnBehalfTo "user@domain.com"

    🗑️ מחיקת תיבת דואר (Soft Delete)

    # מחיקת תיבת דואר
Remove-Mailbox -Identity "user@domain.com"

# ⚠️ פעולה זו מוחקת את התיבה אך מאפשרת שחזור (Soft Deleted) למשך 30 יום.

    📤 ייצוא רשומות יומן פעילות עבור משתמש מסוים

    # -----------------------------------------------
# 🔍 Export Audit Log Entries for Specific User
# 📄 סקריפט זה מחפש ביומן הפעילות (Unified Audit Log)
#     את כל הפעולות שבוצעו על המשתמש: oron@example.co.il
#     ומייצא את תוצאות השינויים (רישוי, תכונות וכו') לקובץ טקסט.
# 🛟 הפלט נשמר בקובץ: C:\PS\audit_results.txt
# -----------------------------------------------

$outputPath = "C:\PS\audit_results.txt"

$logs | Where-Object { $_.AuditData -like "*oron@example.co.il*" } | ForEach-Object {
    $data = ($_ | Select-Object -ExpandProperty AuditData) | ConvertFrom-Json

    $changes = $data.ModifiedProperties | ForEach-Object {
        " - $($_.Name): `n     From: $($_.OldValue)`n     To:   $($_.NewValue)`n"
    }

    $text = @"
===============================
Time: $($data.CreationTime)
Operation: $($data.Operation)
Performed By: $($data.UserId)
Target User: $($data.ObjectId)
Changes:
$($changes -join "`n")
===============================
"@

    Add-Content -Path $outputPath -Value $text
}

Write-Output "Audit results exported to: $outputPath"
